// 这里注入了一个Fisher–Yates Shuffle算法用来洗牌
jQuery.prototype.shuffle = function () {
    var res = this;
    for (var i = res.length - 1; i >= 0; i--) {
        var rnd = Math.floor(Math.random() * (i + 1));
        var ind = res[rnd]; res[rnd] = res[i]; res[i] = ind;
    }
    return res;
}
